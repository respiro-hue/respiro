function buf(ctx,tp){const l=3*ctx.sampleRate,b=ctx.createBuffer(1,l,ctx.sampleRate),d=b.getChannelData(0);if(tp==='branco'){for(let i=0;i<l;i++)d[i]=Math.random()*2-1}else{let a=0,b1=0,c=0,e=0,f=0,g=0,h=0;for(let i=0;i<l;i++){const w=Math.random()*2-1;a=0.99886*a+w*0.0555179;b1=0.99332*b1+w*0.0750759;c=0.969*c+w*0.153852;e=0.8665*e+w*0.3104856;f=0.55*f+w*0.5329522;g=-0.7616*g-w*0.016898;d[i]=(a+b1+c+e+f+g+h+w*0.5362)*0.11;h=w*0.115926}}return b}
function loop(ctx,b,d){const s=ctx.createBufferSource();s.buffer=b;s.loop=true;s.connect(d);s.start();return s}
